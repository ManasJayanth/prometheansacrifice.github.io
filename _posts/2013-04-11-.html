#include <iostream>
#include <cstdio>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

using namespace std;

int main()
{
pid_t c1,rc1;
int status;
c1 = fork();
switch(c1)
{
case -1:
perror("fork:");
return -1;

case 0:
cout << "Child process here:\n";
cout << "My pid: " << getpid() << "\n";
cout << "My Parent's pid: " << getppid() << "\n";
cout << "Exiting...\n";
_exit(4);

default:
rc1 = wait(&status);
cout << "Parent here:\n";
cout << "My pid: " << getpid() << "\n";
cout << "Child pid: " << c1 << "\n";
cout << "Return value from wait(): " << rc1 << "\n";
cout << "Child's exit status: " << WEXITSTATUS(status) << "\n";
}
return 0;
}
